<script src="https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js"></script>
<script>
  // CONFIGURACIÓN: PEGA AQUÍ TU URL DE GOOGLE SHEETS (PUBLICA COMO CSV)
  const SHEET_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSGjOja6tEOQ7pMgNbnGiPVpyFzs-fwwMyw3Yw6HMTPTZWEAAktypqNCp8VSWefog/pub?output=csv';

  let currentRankings = [];

  // Al cargar la página, intenta traer los datos automáticamente
  window.addEventListener('DOMContentLoaded', () => {
    if (SHEET_CSV_URL && SHEET_CSV_URL !== 'TU_URL_DE_GOOGLE_SHEETS_AQUI') {
      fetchData();
    } else {
      document.getElementById('status-msg').innerText = "Configura la URL de Google Sheets en el código.";
      document.getElementById('uploadPanel').style.display = 'block';
    }
  });

  // Escuchador para subida manual por si acaso
  document.getElementById('fileInput').addEventListener('change', (e) => {
    const file = e.target.files[0];
    const reader = new FileReader();
    reader.onload = (evt) => {
      const data = evt.target.result;
      processWorkbook(XLSX.read(data, { type: 'binary' }));
    };
    reader.readAsBinaryString(file);
  });

  async function fetchData() {
    try {
      const response = await fetch(SHEET_CSV_URL);
      const csvText = await response.text();
      const workbook = XLSX.read(csvText, { type: 'string' });
      processWorkbook(workbook);
      document.getElementById('status-msg').innerText = "Datos actualizados automáticamente";
    } catch (e) {
      console.error(e);
      document.getElementById('status-msg').innerText = "Error en carga automática. Usa la subida manual.";
      document.getElementById('uploadPanel').style.display = 'block';
    }
  }

  function processWorkbook(workbook) {
    const sheet = workbook.Sheets[workbook.SheetNames[0]];
    const rawData = XLSX.utils.sheet_to_json(sheet);
    
    // Aquí usamos tu lógica original de procesamiento
    currentRankings = rawData.map(row => ({
      name: row.Tienda || row.Nombre || "Tienda",
      ventas: parseFloat(row.Ventas || 0),
      tickets: parseInt(row.Tickets || 0),
      indicador: parseFloat(row.Indicador || 0)
    })).sort((a, b) => b.indicador - a.indicador);

    renderDashboard();
  }

  function renderDashboard() {
    document.getElementById('statsGrid').style.display = 'grid';
    document.getElementById('mainContent').style.display = 'block';
    
    const tbody = document.getElementById('tableBody');
    let totalV = 0, totalT = 0, sumaInd = 0;

    tbody.innerHTML = currentRankings.map((tienda, index) => {
      totalV += tienda.ventas;
      totalT += tienda.tickets;
      sumaInd += tienda.indicador;
      const pct = Math.min(tienda.indicador * 100, 100);

      return `
        <tr>
          <td>#${index + 1}</td>
          <td style="font-weight:600">${tienda.name}</td>
          <td>${tienda.ventas.toLocaleString()}€</td>
          <td>${tienda.tickets}</td>
          <td style="color:var(--gold); font-weight:bold">${(tienda.indicador * 100).toFixed(2)}%</td>
          <td>
            <div class="bar-bg"><div class="bar-fill" style="width:${pct}%"></div></div>
          </td>
        </tr>
      `;
    }).join('');

    document.getElementById('totalVentas').innerText = totalV.toLocaleString() + '€';
    document.getElementById('totalTickets').innerText = totalT.toLocaleString();
    document.getElementById('mediaIndicador').innerText = (sumaInd / currentRankings.length * 100).toFixed(2) + '%';
  }
</script>

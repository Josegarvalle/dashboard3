<script>
  // 1. COLOCA TU URL DE GOOGLE SHEETS AQUÍ (Publicado como CSV)
  const CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSGjOja6tEOQ7pMgNbnGiPVpyFzs-fwwMyw3Yw6HMTPTZWEAAktypqNCp8VSWefog/pub?output=csv';

  let rawData = [];
  let currentView = 'diario';

  // Carga automática al abrir
  window.addEventListener('DOMContentLoaded', () => {
    if(CSV_URL.includes('TU_URL')) {
        document.getElementById('loading-status').innerText = "⚠️ Error: Falta la URL del Excel";
    } else {
        fetchData();
    }
  });

  async function fetchData() {
    try {
      const response = await fetch(CSV_URL);
      const data = await response.text();
      const workbook = XLSX.read(data, { type: 'string' });
      const sheet = workbook.Sheets[workbook.SheetNames[0]];
      rawData = XLSX.utils.sheet_to_json(sheet);
      
      document.getElementById('loading-status').style.display = 'none';
      renderTable();
    } catch (e) {
      document.getElementById('loading-status').innerText = "❌ Error al cargar datos";
    }
  }

  function setView(view) {
    currentView = view;
    document.getElementById('btnDiario').classList.toggle('active', view === 'diario');
    document.getElementById('btnAcum').classList.toggle('active', view === 'acumulado');
    document.getElementById('labelVentas').innerText = view === 'diario' ? 'Ventas' : 'Ventas Acum.';
    document.getElementById('labelTickets').innerText = view === 'diario' ? 'Tickets' : 'Tickets Acum.';
    renderTable();
  }

  function renderTable(filterText = '') {
    const tbody = document.getElementById('tableBody');
    
    // Procesar datos según la vista
    let processed = rawData.map(row => {
        return {
            name: row.Tienda || row.Nombre,
            ventas: currentView === 'diario' ? parseFloat(row.Ventas || 0) : parseFloat(row['Ventas Acum'] || 0),
            tickets: currentView === 'diario' ? parseInt(row.Tickets || 0) : parseInt(row['Tickets Acum'] || 0),
            indicador: currentView === 'diario' ? parseFloat(row.Indicador || 0) : parseFloat(row['Indicador Acum'] || 0)
        };
    });

    // Ordenar por indicador
    processed.sort((a, b) => b.indicador - a.indicador);

    // Filtrar por texto
    if(filterText) {
        processed = processed.filter(p => p.name.toLowerCase().includes(filterText.toLowerCase()));
    }

    tbody.innerHTML = processed.map((item, index) => {
        const pct = Math.min(item.indicador * 100, 100);
        return `
            <tr>
                <td>#${index + 1}</td>
                <td><strong>${item.name}</strong></td>
                <td>${item.ventas.toLocaleString()}€</td>
                <td>${item.tickets}</td>
                <td style="color:var(--gold)">${(item.indicador * 100).toFixed(2)}%</td>
                <td>
                    <div class="bar-bg"><div class="bar-fill" style="width:${pct}%"></div></div>
                </td>
            </tr>
        `;
    }).join('');
  }

  function filterData() {
    const text = document.getElementById('searchInput').value;
    renderTable(text);
  }
